  # Upload dokumentu pracownika
  curl -X POST "http://localhost:8080/api/files/documents/jan.kowalski@example.com" \
       -H "Content-Type: multipart/form-data" \
       -F "file=@/ścieżka/do/pliku.pdf"

  # Pobranie listy dokumentów pracownika
  curl -X GET "http://localhost:8080/api/files/documents/jan.kowalski@example.com"

  # Pobranie konkretnego dokumentu
  curl -X GET "http://localhost:8080/api/files/documents/jan.kowalski@example.com/123" \
       -o dokument.pdf

  # Usunięcie dokumentu pracownika
  curl -X DELETE "http://localhost:8080/api/files/documents/jan.kowalski@example.com/123"

  # Import pracowników z pliku CSV
  curl -X POST "http://localhost:8080/api/files/import/csv" \
       -H "Content-Type: multipart/form-data" \
       -F "file=@/ścieżka/do/pracownicy.csv"

  # Import pracowników z pliku XML
  curl -X POST "http://localhost:8080/api/files/import/xml" \
       -H "Content-Type: multipart/form-data" \
       -F "file=@/ścieżka/do/pracownicy.xml"

  # Eksport pracowników do pliku CSV (opcjonalnie z parametrem company)
  curl -X GET "http://localhost:8080/api/files/export/csv?company=FirmaX" \
       -o pracownicy.csv

  # Generowanie raportu PDF dla firmy
  curl -X GET "http://localhost:8080/api/files/reports/statistics/TechCorp" -o raport.pdf
